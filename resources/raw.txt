
def showDigit(int a) -> void {
    putChar(a+48);
}

def recursivePrint(int a)-> void {
    if (a!=0) {
         int b = a % 10;
         a = a/10;
         recursivePrint(a);
         showDigit(b);
    } else {}
}

def print(int a)->void {
    if (a==0) {
        showDigit(0);
    } else {}

    recursivePrint(a);
}

def printLn(int a) -> void {
    print(a);
    putChar(11);
}

def main()->int{
    printLn(1);
}
