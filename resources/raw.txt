class Bool [] {
    int data;
}

def Bool_(int d) -> Bool {
    Bool res;
    if (d == 0) {
        res.data = 0;
    } else {
        res.data = 1;
    }
    return res;
}

def and_(Bool a, Bool b) -> Bool {
    return Bool_(a.data & b.data);
}

def main() -> int {
    Bool a = Bool_(1);
    Bool b = Bool_(0);
    Bool res = and_(a, b);
    return res.data;
}
