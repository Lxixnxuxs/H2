        Node n;
        n.construct(v);

        Node res = head.next;

    def getNode(int nr) -> Node {
        if (nr >= size) {printError(9902);} else {}
            /*return the error code for out of bounds*/

        Node cur = head;

            nr = nr - 1;
            cur = cur.next;
        }
        return cur;
    }

        Node n = this.getNode(nr);
        return n.value;

def exp(int base, int exp_) -> int {
    int res = 1;
    while(exp_>0){
        exp_ = exp_ - 1;
        res = res * base;
    }
    while (exp_ < 0) {
        exp_ = exp_ + 1;
        res = res / base;

}
