def h(int n) -> int /% _O(n) %/  {
    while (n>0) /% _O(n) %/  {
        n = (n - 1);
    }
}


def f(int n) -> void /% _O(n) %/  {
    if (n<=0) {
        return 1;
    } else {
        int res = h[n];
        res = (res + f[(n / 2)]);
        return res;
    }
}
